\section*{Questions}

\begin{enumerate}
\item 
\begin{enumerate}
    \item Write the relationship between density $\rho$, mass $m$ and volume $V$.
    \item Rearrange the equation to make (i) $m$ and (ii) $V$ the subject.
    \item State the SI units of density, mass and volume.
\end{enumerate}

% ---------- Regular Questions ----------
\item A sample has a mass of \SI{2.5}{\kilogram} and occupies a volume of \SI{0.0025}{\metre\cubed}. Calculate its density.
\item A metal cube has a mass of \SI{0.35}{\kilogram} and a volume of \SI{5.0e-5}{\metre\cubed}. Calculate its density.
\item A liquid has a density of \SI{800}{\kilogram\per\metre\cubed}. What volume does \SI{4}{\kilogram} of the liquid occupy?
\item Ice has a density of \SI{920}{\kilogram\per\metre\cubed}. What is the mass of \SI{0.015}{\metre\cubed} of ice?
\item A block of material has a mass of \SI{1.2}{\kilogram} and a volume of \SI{0.0013}{\metre\cubed}. Determine its density.
\item Aluminium has a density of \SI{2700}{\kilogram\per\metre\cubed}. What is the mass of \SI{0.004}{\metre\cubed} of aluminium?
\item A sample with a mass of \SI{0.60}{\kilogram} has a density of \SI{600}{\kilogram\per\metre\cubed}. Calculate its volume.
\item Gold has a density of \SI{19300}{\kilogram\per\metre\cubed}. Find the mass of a piece of gold with a volume of \SI{7.5e-4}{\metre\cubed}.
\item A liquid sample has a mass of \SI{15}{\gram} and a volume of \SI{1.3e-5}{\metre\cubed}. Calculate its density.
\item Cooking oil has a density of \SI{900}{\kilogram\per\metre\cubed}. What volume is occupied by \SI{2.7}{\kilogram} of oil?
\item A gas occupies \SI{0.10}{\metre\cubed} and has a mass of \SI{0.125}{\kilogram}. Determine its density.
\item Steel has a density of \SI{7850}{\kilogram\per\metre\cubed}. What volume does a \SI{25}{\kilogram} block of steel occupy?
\item Aluminium has a density of \SI{2700}{\kilogram\per\metre\cubed}. Calculate the mass of \SI{0.002}{\metre\cubed} of aluminium.
\item A liquid sample has a mass of \SI{3.5}{\kilogram} and occupies \SI{0.004}{\metre\cubed}. Find its density.
\item Water has a density of \SI{1000}{\kilogram\per\metre\cubed}. Determine the volume that \SI{0.5}{\kilogram} of water will occupy.

% ---------- Extension Questions ----------
\item \textbf{Extension.} A pine block measuring \SI{0.20}{\metre} $\times$ \SI{0.10}{\metre} $\times$ \SI{0.05}{\metre} has a mass of \SI{750}{\gram}. Calculate its density in \si{\kilogram\per\metre\cubed}. (You must convert the mass to kilograms.)
\item A solid aluminium cylinder has a radius of \SI{0.04}{\metre} and a height of \SI{0.15}{\metre}. Given the density of aluminium is \SI{2700}{\kilogram\per\metre\cubed}, determine its mass.
\item A cube of brass (density \SI{8500}{\kilogram\per\metre\cubed}) has a mass of \SI{1.7}{\kilogram}. Calculate the length of one side of the cube.
\item A slab of marble measuring \SI{0.50}{\metre} $\times$ \SI{0.30}{\metre} $\times$ \SI{0.020}{\metre} is cut from a larger block. Find its mass, taking the density of marble as \SI{2700}{\kilogram\per\metre\cubed}.
\item A solid steel sphere with radius \SI{0.06}{\metre} is cast. Calculate its mass. (Density of steel: \SI{7850}{\kilogram\per\metre\cubed}.)
\end{enumerate}

\section*{Answer Key}
\begin{enumerate}
\item 
\begin{enumerate}
    \item $\rho = \dfrac{m}{V}$
    \item $m = \rho V \qquad\qquad V = \dfrac{m}{\rho}$
    \item Density: \si{\kilogram\per\metre\cubed}; Mass: \si{\kilogram}; Volume: \si{\metre\cubed}
\end{enumerate}

\item $\rho = \SI{1000}{\kilogram\per\metre\cubed}$
\item $\rho = \num{7.0e3}\,\si{\kilogram\per\metre\cubed}$
\item $V = \SI{5.0e-3}{\metre\cubed}$
\item $m = \SI{13.8}{\kilogram}$
\item $\rho \approx \SI{9.23e2}{\kilogram\per\metre\cubed}$
\item $m = \SI{10.8}{\kilogram}$
\item $V = \SI{1.0e-3}{\metre\cubed}$
\item $m = \SI{14.5}{\kilogram}$
\item $\rho \approx \SI{1.15e3}{\kilogram\per\metre\cubed}$
\item $V = \SI{3.0e-3}{\metre\cubed}$
\item $\rho = \SI{1.25}{\kilogram\per\metre\cubed}$
\item $V \approx \SI{3.18e-3}{\metre\cubed}$
\item $m = \SI{5.4}{\kilogram}$
\item $\rho = \SI{8.75e2}{\kilogram\per\metre\cubed}$
\item $V = \SI{5.0e-4}{\metre\cubed}$

% Extension answers
\item $\rho = \SI{7.5e2}{\kilogram\per\metre\cubed}$
\item $m \approx \SI{2.04}{\kilogram}$
\item $\text{Side length} \approx \SI{0.058}{\metre}\, (\text{or } \SI{5.8}{\centi\metre})$
\item $m = \SI{8.10}{\kilogram}$
\item $m \approx \SI{7.10}{\kilogram}$
\end{enumerate}
