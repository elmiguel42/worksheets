\begin{center}
    \Large \textbf{Elastic Potential Energy and Hooke's Law}
\end{center}

\[
E = \frac{1}{2}ke^2
\qquad\qquad
F = -ke
\]

\begin{enumerate}[label=\textbf{\arabic*.}]

\item A spring has a spring constant of \SI{120}{N/m} and is extended by \SI{0.05}{m}. Calculate the force exerted by the spring.

\item A spring exerts a force of \SI{18}{N} when extended by \SI{0.12}{m}. Calculate the spring constant.

\item A spring has a spring constant of \SI{250}{N/m}. Calculate the extension produced by a force of \SI{40}{N}.

\item A spring is compressed by \SI{0.03}{m}. Its spring constant is \SI{600}{N/m}. Calculate the restoring force.

\item A spring extends by \SI{8}{cm} when a force of \SI{24}{N} is applied. Calculate the spring constant.

\item A spring with spring constant \SI{200}{N/m} is extended by \SI{0.10}{m}. Calculate the elastic potential energy stored.

\item A spring stores \SI{0.75}{J} of elastic potential energy when extended by \SI{0.05}{m}. Calculate the spring constant.

\item A spring has a spring constant of \SI{150}{N/m} and stores \SI{3.0}{J} of elastic potential energy. Calculate the extension.

\item A spring is extended by \SI{0.20}{m} and stores \SI{5.0}{J} of elastic potential energy. Calculate the spring constant.

\item A spring has a spring constant of \SI{80}{N/m}. Calculate the elastic potential energy stored when it is extended by \SI{15}{cm}.

\item A force of \SI{30}{N} extends a spring by \SI{0.12}{m}. Calculate the spring constant, then calculate the elastic potential energy stored.

\item A spring has a spring constant of \SI{500}{N/m}. Calculate the force needed to extend it by \SI{0.04}{m}, then calculate the elastic potential energy stored.

\item A spring stores \SI{2.4}{J} of elastic potential energy when extended by \SI{0.08}{m}. Calculate the spring constant and the force required to produce this extension.

\item A spring has a spring constant of \SI{300}{N/m}. It is stretched from \SI{2}{cm} extension to \SI{10}{cm} extension. Calculate the increase in elastic potential energy.

\item A spring is extended by a force of \SI{45}{N}. The spring constant is \SI{180}{N/m}. Calculate the extension and the elastic potential energy stored.

\end{enumerate}

\newpage

\begin{center}
    \Large \textbf{Answer Key}
\end{center}

\begin{enumerate}[label=\textbf{\arabic*.}]

\item
\[
F = ke = 120 \times 0.05 = \SI{6.0}{N}
\]

\item
\[
k = \frac{F}{e} = \frac{18}{0.12} = \SI{150}{N/m}
\]

\item
\[
e = \frac{F}{k} = \frac{40}{250} = \SI{0.16}{m}
\]

\item
\[
F = ke = 600 \times 0.03 = \SI{18}{N}
\]

\item
\[
e = \SI{8}{cm} = \SI{0.08}{m}
\]
\[
k = \frac{F}{e} = \frac{24}{0.08} = \SI{300}{N/m}
\]

\item
\[
E = \frac{1}{2} \times 200 \times 0.10^2 = \SI{1.0}{J}
\]

\item
\[
k = \frac{2E}{e^2} = \frac{2 \times 0.75}{0.05^2} = \SI{600}{N/m}
\]

\item
\[
e = \sqrt{\frac{2E}{k}} = \sqrt{\frac{2 \times 3.0}{150}} = \SI{0.20}{m}
\]

\item
\[
k = \frac{2E}{e^2} = \frac{2 \times 5.0}{0.20^2} = \SI{250}{N/m}
\]

\item
\[
e = \SI{0.15}{m}
\]
\[
E = \frac{1}{2} \times 80 \times 0.15^2 = \SI{0.90}{J}
\]

\item
\[
k = \frac{30}{0.12} = \SI{250}{N/m}
\]
\[
E = \frac{1}{2} \times 250 \times 0.12^2 = \SI{1.8}{J}
\]

\item
\[
F = 500 \times 0.04 = \SI{20}{N}
\]
\[
E = \frac{1}{2} \times 500 \times 0.04^2 = \SI{0.40}{J}
\]

\item
\[
k = \frac{2 \times 2.4}{0.08^2} = \SI{750}{N/m}
\]
\[
F = 750 \times 0.08 = \SI{60}{N}
\]

\item
\[
\Delta E = \frac{1}{2} \times 300 \times 0.10^2 - \frac{1}{2} \times 300 \times 0.02^2 = \SI{1.44}{J}
\]

\item
\[
e = \frac{45}{180} = \SI{0.25}{m}
\]
\[
E = \frac{1}{2} \times 180 \times 0.25^2 = \SI{5.625}{J}
\]

\end{enumerate}
