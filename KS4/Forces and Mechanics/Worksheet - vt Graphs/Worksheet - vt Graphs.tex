\section*{Mark Scheme (Velocity--Time Graph Worksheet)}

\textbf{Note:} 
\begin{itemize}
  \item Constant velocity = horizontal section on a $v$--$t$ graph
  \item Acceleration = gradient ($\Delta v / \Delta t$)
  \item Distance travelled = area under the graph
\end{itemize}

% ===================== GRAPH 1 =====================
\subsection*{Graph 1}

\begin{enumerate}[label=\textbf{1.\arabic*}]
  \item Constant velocity: \textbf{15--25 s} (at $5\,\si{m/s}$).

  \item Maximum velocity: \textbf{$5\,\si{m/s}$}, first reached at \textbf{15 s}.

  \item Greatest acceleration: \textbf{5--10 s} and \textbf{10--15 s} 
  (both have the steepest positive gradient, $0.4\,\si{m/s^2}$).

  \item Acceleration 5--15 s:
  \[
  a=\frac{5-1}{15-5}=\frac{4}{10}=\textbf{0.4\,\si{m/s^2}}
  \]

  \item Distance travelled 15--25 s:
  \[
  s = 5 \times 10 = \textbf{50\,\si{m}}
  \]

  \item Motion 25--40 s (any sensible set of bullet points describing the trend earns credit), e.g.
  \begin{itemize}
    \item Still moving forwards (velocity stays positive)
    \item Slows down from $5$ to $4\,\si{m/s}$ (25--30 s)
    \item Slows down from $4$ to $2\,\si{m/s}$ (30--35 s)
    \item Slows down from $2$ to $1\,\si{m/s}$ (35--40 s)
  \end{itemize}
\end{enumerate}

% ===================== GRAPH 2 =====================
\subsection*{Graph 2}

\begin{enumerate}[label=\textbf{2.\arabic*}]
  \item First constant velocity interval: \textbf{15--20 s} (at $5\,\si{m/s}$).

  \item $v(10\,\si{s})=\textbf{4\,\si{m/s}}$ \quad and \quad $v(30\,\si{s})=\textbf{2\,\si{m/s}}$.

  \item Accelerating most rapidly: \textbf{0--5 s} and \textbf{5--10 s}
  (steepest positive gradient, $0.4\,\si{m/s^2}$).  
  (Also acceptable: \textbf{20--25 s} as the most rapid \emph{deceleration}, $-0.4\,\si{m/s^2}$.)

  \item Acceleration 0--10 s:
  \[
  a=\frac{4-0}{10-0}=\frac{4}{10}=\textbf{0.4\,\si{m/s^2}}
  \]

  \item Distance travelled 35--45 s:
  \[
  \text{35--40: } \frac{(2+1)}{2}\times 5 = 7.5\,\si{m}
  \qquad
  \text{40--45: } \frac{(1+0)}{2}\times 5 = 2.5\,\si{m}
  \]
  Total $=\textbf{10\,\si{m}}$.

  \item Whole journey description (any sensible bullet summary earns credit), e.g.
  \begin{itemize}
    \item Starts from rest and speeds up to $5\,\si{m/s}$ by 15 s
    \item Travels at constant $5\,\si{m/s}$ from 15--20 s
    \item Slows down after 20 s
    \item Travels at constant $2\,\si{m/s}$ from 30--35 s
    \item Slows to rest by 45 s
  \end{itemize}
\end{enumerate}
