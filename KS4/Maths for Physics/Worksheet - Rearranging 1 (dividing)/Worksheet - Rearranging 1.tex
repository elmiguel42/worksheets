% =========================
% Worked Example 1
% =========================
\textbf{Worked Example 1}\\
\textbf{Question:} \( y = mx \), make \( x \) the subject.

\[
y = mx
\]

\begin{itemize}[leftmargin=*, itemsep=2pt]
    \item \(x\) is positive and a numerator $\rightarrow$ leave it where it is.
    \item Current operation \(= \times m\)
    \item Inverse operation \(= \div m\)
    \item Apply the inverse to BOTH sides:
\end{itemize}

\[
\frac{y}{m}=\frac{mx}{m}
\]

\textbf{Cancel:}
\[
\frac{y}{m}=\frac{\cancel{m}x}{\cancel{m}}
\]

\textbf{Final solution:}
\[
x=\frac{y}{m}
\]
